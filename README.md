    #of the first list. 
    #Sample List : [1,2,3,3,3,3,4,5]
    #Unique List : [1, 2, 3, 4, 5]

    def uniq(number):
      x = []
      for a in number:
        if a not in x:
          x.append(a)
      return x

    print(uniq([1,2,3,3,3,3,4,5]))
